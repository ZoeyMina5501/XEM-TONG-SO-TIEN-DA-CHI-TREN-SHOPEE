                                    var amount = item?.amount || 
                                               item?.quantity || 0;
                                    
                                    tongSanPhamDaMua += amount;

        } else if (this.readyState == 4) {

}

function formatPrice(number, fixed = 0) {
    if (isNaN(number)) return '0';
